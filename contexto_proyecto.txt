  if (user) {
    userRole.value = user.user_metadata?.rol || 'paciente'

    const nombres = user.user_metadata?.nombres || ''
    const apellidos = user.user_metadata?.apellidos || ''
    if (nombres || apellidos) {
      displayName.value = `${nombres} ${apellidos}`.trim()
    } else {
      displayName.value = user.email.split('@')[0]
    }
  }
}

  // ðŸ“¢ Banner de notificaciÃ³n clÃ¡sico (Auto-cerrable)
  const showAlert = (msg, alertType = 'info') => {

    isVisible.value = true

    timeoutId.value = setTimeout(() => {

    }, 4000)
  }

  const closeAlert = () => {

    if (timeoutId.value) clearTimeout(timeoutId.value)
  }

  const showConfirm = (msg) => {
    confirmMessage.value = msg
    isConfirmVisible.value = true
    
    // Retornamos una promesa que se resolverÃ¡ cuando el usuario dÃ© clic en Aceptar o Cancelar
    return new Promise((resolve) => {
      confirmResolve = resolve
    })
  }

  // Manejadores de respuesta del Modal
  const acceptConfirm = () => {
    isConfirmVisible.value = false
    if (confirmResolve) confirmResolve(true) // Retorna true
  }

  const cancelConfirm = () => {
    isConfirmVisible.value = false
    if (confirmResolve) confirmResolve(false) // Retorna false
  }

  return {
    // Retornos del banner
    isVisible,
    message,
    type,
    showAlert,
    closeAlert,
    
    // Retornos de la confirmaciÃ³n
    isConfirmVisible,
    confirmMessage,
    showConfirm,
    acceptConfirm,

  }

    // â”€â”€ fetchCitas â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const fetchCitas = async (filtros = {}) => {
        loading.value = true
        try {
            // Consultamos la vista unificada
            let query = supabase.from('vw_citas_detalladas').select(`
            *,
            Fisioterapeuta (
                idFisioterapeuta, especialidad,
                Persona (nombres, apellidos)
            ),
            Paciente (
                idPaciente,
                Persona (nombres, apellidos, celular)
            )
        `)

            if (esPaciente.value) query = query.eq('idPaciente', userId.value)
            else if (esFisioterapeuta.value) query = query.eq('idFisioterapeuta', userId.value)

            if (filtros.rangoInicio && filtros.rangoFin) {
                query = query.gte('fecha_hora', filtros.rangoInicio).lte('fecha_hora', filtros.rangoFin)
            } else if (filtros.fecha) {
                const inicio = `${filtros.fecha}T00:00:00+00:00`
                const fin = `${filtros.fecha}T23:59:59+00:00`
                query = query.gte('fecha_hora', inicio).lte('fecha_hora', fin)
            }
            if (filtros.estado) query = query.eq('estado', filtros.estado)
            if (filtros.idFisioterapeuta) query = query.eq('idFisioterapeuta', filtros.idFisioterapeuta)

            const { data, error } = await query.order('fecha_hora', { ascending: true })
            if (error) throw error
            citas.value = data ?? []
        } catch (err) {
            showAlert('Error al cargar citas: ' + err.message, 'error')
        } finally {
            loading.value = false
        }
    }

    // â”€â”€ fetchFisios â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const fetchFisios = async () => {
        const { data, error } = await supabase
            .from('Fisioterapeuta')
            .select(`
        idFisioterapeuta,
        especialidad,
        Persona ( nombres, apellidos )
      `)
        if (error) { showAlert('Error al cargar fisioterapeutas: ' + error.message, 'error'); return }
        fisios.value = data ?? []
    }

    // â”€â”€ fetchPacientes â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const fetchPacientes = async () => {
        const { data, error } = await supabase
            .from('Paciente')
            .select(`
        idPaciente,
        Persona ( nombres, apellidos, celular, numero_documento )
      `)
        if (error) { showAlert('Error al cargar pacientes: ' + error.message, 'error'); return }
        pacientes.value = data ?? []
    }

    // â”€â”€ fetchPaquetesPaciente â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const fetchPaquetesPaciente = async (idPaciente) => {
        if (!idPaciente) { paquetes.value = []; return }
        try {
            const { data, error } = await supabase
                .from('v_saldo_paquete')
                .select('idPaquete, nombre_paquete, total_sesiones, sesiones_consumidas, sesiones_restantes, estado_pago')
                .eq('idPaciente', idPaciente)
                .gt('sesiones_restantes', 0)
                .order('idPaquete', { ascending: false })

            if (error) throw error
            paquetes.value = data ?? []
        } catch (err) {
            showAlert('Error al cargar paquetes del paciente: ' + err.message, 'error')
        }
    }

    // â”€â”€ fetchHorarioFisio â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const fetchHorarioFisio = async (idFisioterapeuta) => {
        if (!idFisioterapeuta) { horarios.value = []; return }
        const { data, error } = await supabase
            .from('Horario')
            .select('dia_semana, hora_inicio, hora_fin')
            .eq('idFisioterapeuta', idFisioterapeuta)
        if (error) { showAlert('Error al cargar horario: ' + error.message, 'error'); return }
        horarios.value = data ?? []
    }

    // â”€â”€ obtenerSlotsDisponibles â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const obtenerSlotsDisponibles = async (idFisioterapeuta, fechaSeleccionada, duracionMinutos) => {
        if (!idFisioterapeuta || !fechaSeleccionada) return []

        const [yyyy, mm, dd] = fechaSeleccionada.split('-')
        const fechaLocal = new Date(yyyy, mm - 1, dd)
        let diaSemana = fechaLocal.getDay()
        diaSemana = diaSemana === 0 ? 7 : diaSemana

        const { data: horariosFisio } = await supabase
            .from('Horario')
            .select('hora_inicio, hora_fin')
            .eq('idFisioterapeuta', idFisioterapeuta)
            .eq('dia_semana', diaSemana)

        if (!horariosFisio || horariosFisio.length === 0) return []

        const inicioDiaIso = new Date(yyyy, mm - 1, dd, 0, 0, 0).toISOString()
        const finDiaIso = new Date(yyyy, mm - 1, dd, 23, 59, 59).toISOString()

        const { data: citasDelDia } = await supabase
            .from('Sesion')
            .select('fecha_hora, tipo')
            .eq('idFisioterapeuta', idFisioterapeuta)
            .gte('fecha_hora', inicioDiaIso)
            .lte('fecha_hora', finDiaIso)
            .not('estado', 'in', '(cancelada,reprogramada,no_asistio)')

        const citasOcupadas = (citasDelDia || []).map(cita => {
            const fc = new Date(cita.fecha_hora)
            const inicioMinutos = fc.getHours() * 60 + fc.getMinutes()
            const duracionCita = cita.tipo === 'evaluacion' ? 20 : 60
            return { inicio: inicioMinutos, fin: inicioMinutos + duracionCita }
        })

        const parseTime = (timeStr) => {
            const [h, m] = timeStr.split(':').map(Number)
            return h * 60 + m
        }

        const formatTime = (mins) => {
            return `${String(Math.floor(mins / 60)).padStart(2, '0')}:${String(mins % 60).padStart(2, '0')}`
        }

        const slotsLibres = []

        horariosFisio.forEach(bloque => {
            let actual = parseTime(bloque.hora_inicio)
            const finBloque = parseTime(bloque.hora_fin)

            while (actual + duracionMinutos <= finBloque) {
                const finActual = actual + duracionMinutos

                // En lugar de solo saber si hay choque, averiguamos CON CUÃL cita choca
                const choque = citasOcupadas.find(c => actual < c.fin && finActual > c.inicio)

                if (!choque) {
                    slotsLibres.push(formatTime(actual))
                    // Si el espacio estÃ¡ libre, avanzamos el bloque completo (ej: 60 min)
                    // para mostrar 08:00, 09:00, 10:00 y evitar agendas fragmentadas.
                    actual += duracionMinutos
                } else {
                    // Si choca con una cita (ej. una evaluaciÃ³n de 08:00 a 08:20),
                    // saltamos directamente a la hora en que termina esa cita (08:20).
                    actual = choque.fin
                }
            }
        })

        return slotsLibres
    }

    // â”€â”€ crearCita â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const crearCita = async (payload) => {
        const {
            idPaciente, idFisioterapeuta, fecha_hora, modalidad,
            idPaquete = null, idServicio = null, // âœ… Capturamos el nuevo campo
            cantidadSesiones = null, observaciones = null,
        } = payload

        if (!idPaciente) { showAlert('Debe seleccionar un paciente.', 'error'); return false }
        if (!idFisioterapeuta) { showAlert('Debe seleccionar un fisioterapeuta.', 'error'); return false }
        if (!fecha_hora) { showAlert('Debe seleccionar fecha y hora.', 'error'); return false }
        if (!modalidad) { showAlert('Debe seleccionar la modalidad de la cita.', 'error'); return false }


        if (modalidad === 'evaluacion_paquete' && !idPaquete) {
            showAlert('Debe seleccionar el paquete del paciente.', 'error')
            return false
        }
        if (modalidad === 'evaluacion_sesiones_libres' && (!cantidadSesiones || cantidadSesiones < 1)) {
            showAlert('Debe indicar la cantidad de sesiones adicionales (mÃ­nimo 1).', 'error')
            return false
        }

        const fechaCita = new Date(fecha_hora)
        const ahora = new Date()
        ahora.setMinutes(ahora.getMinutes() - 5) // 5 min de gracia

        if (fechaCita < ahora) {
            showAlert('No se puede agendar una cita para una fecha y hora en el pasado.', 'error')
            return false
        }

        const slotOcupado = await verificarDisponibilidad(idFisioterapeuta, fecha_hora)
        if (slotOcupado) {
            showAlert('Ese horario ya estÃ¡ ocupado para el fisioterapeuta seleccionado.', 'error')
            return false
        }

        loadingAccion.value = true
        try {
            const esTratamiento = modalidad === 'sesion_tratamiento'

            const { data: sesion, error: errSesion } = await supabase
                .from('Sesion')
                .insert({
                    idPaciente,
                    idFisioterapeuta,
                    fecha_hora,
                    estado: 'reservada',
                    tipo: esTratamiento ? 'tratamiento' : 'evaluacion',
                    numero_sesion: 1,
                    observaciones,
                    expires_at: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString(),
                    idPaquete,
                    idServicio, // âœ… GUARDAMOS LA RELACIÃ“N CON EL CATÃLOGO
                })
                .select()
                .single()

            if (errSesion) throw errSesion

            showAlert('âœ… Cita registrada exitosamente.', 'success')
            return sesion
        } catch (err) {
            showAlert('Error al registrar: ' + err.message, 'error')
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ registrarCheckIn â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const registrarCheckIn = async (idSesion) => {
        loadingAccion.value = true
        try {
            const { error } = await supabase
                .from('Sesion')
                .update({ paciente_en_sala: true })
                .eq('idSesion', idSesion)
                .eq('estado', 'agendada')
            if (error) throw error
            showAlert('âœ… Paciente registrado en sala de espera.', 'success')
            return true
        } catch (err) {
            showAlert('Error al registrar check-in: ' + err.message, 'error')
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ confirmarPago â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const confirmarPago = async ({ idSesion, idPaciente, monto, metodoPago, numeroOperacion, idPaquete = null }) => {
        if (!metodoPago) { showAlert('Seleccione el mÃ©todo de pago.', 'error'); return false }
        if (monto === null || monto === undefined || monto < 0) { showAlert('Ingrese un monto vÃ¡lido.', 'error'); return false }

        loadingAccion.value = true
        try {
            const { error: errPago } = await supabase
                .from('Pago')
                .insert({
                    idPaciente,
                    idSesion: idPaquete ? null : idSesion,
                    idPaquete: idPaquete || null,
                    monto,
                    metodo_pago: metodoPago,
                    numero_operacion: numeroOperacion || null,
                    estado_pago: 'completado',
                })
            if (errPago) throw errPago

            const { error: errSesion } = await supabase
                .from('Sesion')
                .update({ estado: 'agendada', expires_at: null })
                .eq('idSesion', idSesion)
            if (errSesion) throw errSesion

            showAlert('âœ… Pago confirmado. La cita estÃ¡ agendada.', 'success')
            return true
        } catch (err) {
            if (err.code === 'P0001') {
                showAlert('Regla de negocio: ' + err.message, 'error')
            } else {
                showAlert('Error al confirmar el pago: ' + err.message, 'error')
            }
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ cancelarCita â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const cancelarCita = async ({ idSesion, motivo }) => {
        if (!motivo?.trim()) { showAlert('Ingrese el motivo de la cancelaciÃ³n.', 'error'); return false }
        loadingAccion.value = true
        try {
            const { error } = await supabase
                .from('Sesion')
                .update({ estado: 'cancelada', observaciones: motivo.trim() })
                .eq('idSesion', idSesion)
                .in('estado', ['reservada', 'agendada'])
            if (error) throw error
            showAlert('Cita cancelada correctamente.', 'info')
            return true
        } catch (err) {
            showAlert('Error al cancelar la cita: ' + err.message, 'error')
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ marcarInasistencia â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const marcarInasistencia = async (idSesion) => {
        loadingAccion.value = true
        try {
            const { error } = await supabase
                .from('Sesion')
                .update({ estado: 'no_asistio' })
                .eq('idSesion', idSesion)
                .eq('estado', 'agendada')
            if (error) throw error
            showAlert('Inasistencia registrada.', 'info')
            return true
        } catch (err) {
            showAlert('Error al registrar inasistencia: ' + err.message, 'error')
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ verificarDisponibilidad â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const verificarDisponibilidad = async (idFisioterapeuta, fecha_hora) => {
        const inicio = new Date(fecha_hora)
        const fin = new Date(inicio.getTime() + 60 * 60 * 1000)

        const { data } = await supabase
            .from('Sesion')
            .select('idSesion')
            .eq('idFisioterapeuta', idFisioterapeuta)
            .not('estado', 'in', '(cancelada,reprogramada,no_asistio)')
            .gte('fecha_hora', inicio.toISOString())
            .lt('fecha_hora', fin.toISOString())
            .limit(1)

        return (data?.length ?? 0) > 0
    }

    // â”€â”€ Helpers de UI â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const formatFechaHora = (iso) => {
        if (!iso) return 'â€”'
        return new Intl.DateTimeFormat('es-PE', {
            weekday: 'short', day: '2-digit', month: 'short',
            year: 'numeric', hour: '2-digit', minute: '2-digit', hour12: true,
        }).format(new Date(iso))
    }

    const getEstadoInfo = (estado) =>
        ESTADOS_SESION[estado] ?? { label: estado, color: '#64748b', bg: '#f1f5f9' }

    const nombreCompleto = (persona) => {
        if (!persona) return 'â€”'
        return `${persona.nombres ?? ''} ${persona.apellidos ?? ''}`.trim()
    }


    // â”€â”€ confirmarAsistencia (Para evaluaciones gratuitas) â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const confirmarAsistencia = async (idSesion) => {
        loadingAccion.value = true
        try {
            const { error } = await supabase
                .from('Sesion')
                .update({ estado: 'agendada' })
                .eq('idSesion', idSesion)
                .eq('estado', 'reservada')

            if (error) throw error
            showAlert('âœ… Asistencia confirmada por telÃ©fono. La cita estÃ¡ lista para el paciente.', 'success')
            return true
        } catch (err) {
            showAlert('Error al confirmar asistencia: ' + err.message, 'error')
            return false
        } finally {
            loadingAccion.value = false
        }
    }

    // â”€â”€ reprogramarCita (MÃ¡ximo 1 vez) â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
    const reprogramarCita = async ({ idSesion, idFisioterapeutaNuevo, nuevaFechaHora, motivo }) => {
        if (!motivo?.trim()) { showAlert('Ingrese el motivo de reprogramaciÃ³n.', 'error'); return false }
        if (!nuevaFechaHora) { showAlert('Seleccione la nueva fecha y hora.', 'error'); return false }
        if (!idFisioterapeutaNuevo) { showAlert('Seleccione un fisioterapeuta.', 'error'); return false }

        const fechaCita = new Date(nuevaFechaHora)
        const ahora = new Date()
        ahora.setMinutes(ahora.getMinutes() - 5) // 5 min de gracia

        if (fechaCita < ahora) {
            showAlert('La nueva fecha y hora de reprogramaciÃ³n no puede estar en el pasado.', 'error')
            return false
        }

        loadingAccion.value = true
        try {
            // 1. Obtener los datos exactos de la sesiÃ³n antigua
            const { data: sesionAntigua, error: errGet } = await supabase
                .from('Sesion')
                .select('*')
                .eq('idSesion', idSesion)
                .single()
            if (errGet) throw errGet

            // ðŸ›‘ REGLA DE NEGOCIO: Si ya tiene idSesionOriginal, es que YA fue reprogramada antes.
            if (sesionAntigua.idSesionOriginal !== null) {
                throw new Error('Regla de negocio: Esta cita ya fue reprogramada anteriormente. Solo se permite una vez.')
            }

            // 2. Crear la NUEVA sesiÃ³n clonando los datos de la antigua
            const { data: nuevaSesion, error: errInsert } = await supabase
                .from('Sesion')
                .insert({
                    idPaciente: sesionAntigua.idPaciente,
                    idFisioterapeuta: idFisioterapeutaNuevo,
                    fecha_hora: nuevaFechaHora,
                    estado: sesionAntigua.estado, // Mantiene el estado pagado (agendada) o impago (reservada)
                    tipo: sesionAntigua.tipo,
                    numero_sesion: sesionAntigua.numero_sesion,
                    observaciones: sesionAntigua.observaciones,
                    idPaquete: sesionAntigua.idPaquete,
                    idTratamiento: sesionAntigua.idTratamiento,
